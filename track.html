// ───────────────────────────────────────────────────────────────────────── // Supabase Edge Function: docusign-handler // // Doet 2 dingen (via ?action=... of body.action): // 1) create_envelope — maakt een DocuSign envelope aan met het contract-PDF // en de koper als "embedded" ondertekenaar. // 2) get_signing_url — vraagt de ondertekenlink (recipient view) op voor // een bestaande envelope, om in te bedden in track.html. // // Benodigde Supabase secrets (zet je via: supabase secrets set NAAM=waarde): // DOCUSIGN_INTEGRATION_KEY Integration Key / Client ID uit DocuSign // DOCUSIGN_USER_ID User ID (GUID) van het DocuSign-account dat ondertekent // DOCUSIGN_ACCOUNT_ID Account ID (GUID) // DOCUSIGN_PRIVATE_KEY De volledige inhoud van je RSA private key (-----BEGIN...-----) // DOCUSIGN_BASE_PATH bv. https://demo.docusign.net (sandbox) of https://eu.docusign.net (productie) // DOCUSIGN_AUTH_SERVER bv. account-d.docusign.com (sandbox) of account.docusign.com (productie) // // Deploy met: // supabase functions deploy docusign-handler --no-verify-jwt // (--no-verify-jwt omdat dit door zowel het dashboard als het publieke klantportaal // wordt aangeroepen; de functie doet zelf geen gevoelige acties zonder geldig envelopeId) // ───────────────────────────────────────────────────────────────────────── import jwt from "npm:jsonwebtoken@9.0.2"; const CORS_HEADERS = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type", "Access-Control-Allow-Methods": "POST, OPTIONS", }; function json(body: unknown, status = 200) { return new Response(JSON.stringify(body), { status, headers: { ...CORS_HEADERS, "Content-Type": "application/json" }, }); } // ── JWT-token ophalen bij DocuSign (server-naar-server, geen klant-login nodig) ── async function getAccessToken(): Promise { const integrationKey = Deno.env.get("DOCUSIGN_INTEGRATION_KEY")!; const userId = Deno.env.get("DOCUSIGN_USER_ID")!; const privateKey = Deno.env.get("DOCUSIGN_PRIVATE_KEY")!; const authServer = Deno.env.get("DOCUSIGN_AUTH_SERVER") || "account-d.docusign.com"; const assertion = jwt.sign( { iss: integrationKey, sub: userId, aud: authServer, scope: "signature impersonation", }, privateKey, { algorithm: "RS256", expiresIn: "1h" }, ); const resp = await fetch(`https://${authServer}/oauth/token`, { method: "POST", headers: { "Content-Type": "application/x-www-form-urlencoded" }, body: new URLSearchParams({ grant_type: "urn:ietf:params:oauth:grant-type:jwt-bearer", assertion, }), }); if (!resp.ok) { const errText = await resp.text(); throw new Error(`DocuSign auth mislukt (${resp.status}): ${errText}`); } const data = await resp.json(); return data.access_token as string; } // ── Envelope aanmaken met het contract-PDF, klant als embedded ondertekenaar ── async function createEnvelope(payload: any) { const { pdfBase64, bestandsnaam, koperNaam, koperEmail, dossierId, onderwerp } = payload; if (!pdfBase64 || !koperNaam || !koperEmail) { throw new Error("pdfBase64, koperNaam en koperEmail zijn verplicht."); } const accessToken = await getAccessToken(); const basePath = Deno.env.get("DOCUSIGN_BASE_PATH") || "https://demo.docusign.net"; const accountId = Deno.env.get("DOCUSIGN_ACCOUNT_ID")!; // clientUserId koppelt de ondertekenaar aan "embedded signing" — zonder dit stuurt // DocuSign automatisch een e-mail, en dat willen we hier niet (klant tekent in de portal). const clientUserId = "dossier-" + (dossierId || koperEmail); const envelopeDef = { emailSubject: onderwerp || "Aankoopcontract BLITZ POWER — ter ondertekening", documents: [ { documentBase64: pdfBase64, name: bestandsnaam || "Aankoopcontract.pdf", fileExtension: "pdf", documentId: "1", }, ], recipients: { signers: [ { email: koperEmail, name: koperNaam, recipientId: "1", clientUserId, tabs: { signHereTabs: [ // Ankergebaseerd i.p.v. vaste x/y-coördinaten: DocuSign zoekt zelf naar dit // (onzichtbare) tekstankertje in het document en plaatst het handtekenvakje // er automatisch naast — veel betrouwbaarder dan geschatte pixel-posities. // Het ankertje "/sig_koper/" wordt door het dashboard onzichtbaar in het // contract-PDF gezet, net naast "Handtekening:" bij "Voor Koper" op pagina 2. { anchorString: "/sig_koper/", anchorUnits: "pixels", anchorXOffset: "0", anchorYOffset: "-10", anchorIgnoreIfNotPresent: "false", }, ], }, }, ], }, status: "sent", }; const resp = await fetch( `${basePath}/restapi/v2.1/accounts/${accountId}/envelopes`, { method: "POST", headers: { Authorization: `Bearer ${accessToken}`, "Content-Type": "application/json", }, body: JSON.stringify(envelopeDef), }, ); if (!resp.ok) { const errText = await resp.text(); throw new Error(`Envelope aanmaken mislukt (${resp.status}): ${errText}`); } const data = await resp.json(); return { envelopeId: data.envelopeId, clientUserId }; } // ── Ondertekenlink opvragen voor een bestaande envelope (embedded signing) ── async function getSigningUrl(payload: any) { const { envelopeId, clientUserId, koperNaam, koperEmail, returnUrl } = payload; if (!envelopeId || !clientUserId || !koperNaam || !koperEmail || !returnUrl) { throw new Error("envelopeId, clientUserId, koperNaam, koperEmail en returnUrl zijn verplicht."); } const accessToken = await getAccessToken(); const basePath = Deno.env.get("DOCUSIGN_BASE_PATH") || "https://demo.docusign.net"; const accountId = Deno.env.get("DOCUSIGN_ACCOUNT_ID")!; const viewRequest = { returnUrl, authenticationMethod: "none", email: koperEmail, userName: koperNaam, clientUserId, }; const resp = await fetch( `${basePath}/restapi/v2.1/accounts/${accountId}/envelopes/${envelopeId}/views/recipient`, { method: "POST", headers: { Authorization: `Bearer ${accessToken}`, "Content-Type": "application/json", }, body: JSON.stringify(viewRequest), }, ); if (!resp.ok) { const errText = await resp.text(); throw new Error(`Ondertekenlink ophalen mislukt (${resp.status}): ${errText}`); } const data = await resp.json(); return { signingUrl: data.url }; } // ── Envelope-status opvragen (voor "is dit al ondertekend?") ── async function getEnvelopeStatus(payload: any) { const { envelopeId } = payload; if (!envelopeId) throw new Error("envelopeId is verplicht."); const accessToken = await getAccessToken(); const basePath = Deno.env.get("DOCUSIGN_BASE_PATH") || "https://demo.docusign.net"; const accountId = Deno.env.get("DOCUSIGN_ACCOUNT_ID")!; const resp = await fetch( `${basePath}/restapi/v2.1/accounts/${accountId}/envelopes/${envelopeId}`, { headers: { Authorization: `Bearer ${accessToken}` } }, ); if (!resp.ok) { const errText = await resp.text(); throw new Error(`Status opvragen mislukt (${resp.status}): ${errText}`); } const data = await resp.json(); return { status: data.status }; // "sent" | "delivered" | "completed" | ... } Deno.serve(async (req: Request) => { if (req.method === "OPTIONS") return new Response("ok", { headers: CORS_HEADERS }); try { const body = await req.json(); const action = body.action; if (action === "create_envelope") return json(await createEnvelope(body)); if (action === "get_signing_url") return json(await getSigningUrl(body)); if (action === "get_status") return json(await getEnvelopeStatus(body)); return json({ error: "Onbekende actie. Gebruik create_envelope, get_signing_url of get_status." }, 400); } catch (e: any) { console.error(e); return json({ error: e?.message || String(e) }, 500); } });
